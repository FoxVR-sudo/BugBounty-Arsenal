```python
# detectors/command_injection_detector.py

from detectors.registry import register_active
import asyncio

@register_active
async def command_injection_detector(url, session, config):
    """
    Detect OS command injection vulnerabilities.
    
    Tests for blind and time-based command injection.
    """
    findings = []
    
    # Time-based payloads (sleep detection)
    time_payloads = [
        '; sleep 5',
        '| sleep 5',
        '`sleep 5`',
        '$(sleep 5)',
        '& timeout 5'
    ]
    
    # Parse URL parameters
    from urllib.parse import urlparse, parse_qs, urlencode
    parsed = urlparse(url)
    params = parse_qs(parsed.query)
    
    if not params:
        return findings
    
    for param in params:
        for payload in time_payloads:
            # Measure response time
            start = asyncio.get_event_loop().time()
            
            test_params = params.copy()
            test_params[param] = [payload]
            test_url = f"{parsed.scheme}://{parsed.netloc}{parsed.path}?{urlencode(test_params, doseq=True)}"
            
            try:
                async with session.get(test_url, timeout=10) as resp:
                    elapsed = asyncio.get_event_loop().time() - start
                    
                    # If response took ~5 seconds, likely vulnerable
                    if 4.5 <= elapsed <= 6.0:
                        findings.append({
                            'type': 'Command Injection (Time-Based)',
                            'severity': 'critical',
                            'confidence': 'high',
                            'url': test_url,
                            'parameter': param,
                            'payload': payload,
                            'evidence': f'Response delayed by {elapsed:.2f}s',
                            'impact': 'Remote command execution on server',
                            'recommendation': 'Sanitize user input, use parameterized commands'
                        })
            except:
                continue
    
    return findings
```
